<script>
    // Simular dados do usuário
    const user = {
        status: "Online",
        id: "123456",
        name: "Edson Pereira",
        email: "edson@example.com",
        phone: "+55 11 96479-2220"
    };

    // Simular lista de aluguéis
    const rentals = [
        { name: "Apartamento 101", price: "R$ 1.200,00" },
        { name: "Casa 23", price: "R$ 2.500,00" },
        { name: "Kitnet 5", price: "R$ 900,00" },
    ];

    // Preencher dados do usuário
    document.getElementById("auth-status").textContent = user.status;
    document.getElementById("user-id").textContent = user.id;

    // Preencher lista de aluguéis
    const rentalList = document.getElementById("rental-list");
    rentals.forEach(r => {
        const li = document.createElement("li");
        li.textContent = `${r.name} - ${r.price}`;
        rentalList.appendChild(li);
    });

    // Simular carregamento
    setTimeout(() => {
        document.getElementById("loading").classList.add("hidden");
        document.getElementById("content").classList.remove("hidden");
    }, 1000);

    // Modal do perfil
    const modal = document.getElementById("profile-modal");
    document.getElementById("show-profile-btn").addEventListener("click", () => {
        document.getElementById("profile-name").textContent = user.name;
        document.getElementById("profile-email").textContent = user.email;
        document.getElementById("profile-phone").textContent = user.phone;
        modal.classList.remove("hidden");
    });
    document.getElementById("close-profile").addEventListener("click", () => {
        modal.classList.add("hidden");
    });
</script>
